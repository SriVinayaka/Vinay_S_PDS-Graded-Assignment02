93   import pandas as pd
94   import re

96   def convert_to_inches(height):
97       """Converts a string representation of height in feet and inches to inches. 
98    
99       Args: 
100          height_str (str): A string representation of height in feet and inches, 
101                            e.g., "5' 11", "5 ft 11 in", "5 feet 11 inches", etc. 
102   
103      Returns: 
104          int: The height in inches, or None if the format is invalid. 
105      """
106      height_str = str(height)
107      height_str = height_str.strip().upper()
108  
109      # Regular expression to extract feet and inches
110      match = re.match(r"(\d+)(\'| FT| FEET) (\d+)(\"| IN| INCHES)", height_str)
111      if match:
112          feet, inches = int(match.group(1)), int(match.group(3))
113          return feet * 12 + inches
114  
115      return None
